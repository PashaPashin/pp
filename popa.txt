import random
text="1-2 {комнатные|к.} квартиры в {центре|центре города}. {Хороший|Отличный} ремонт, [пластиковые окна,|кабельное ТВ,|Интернет,|стиральная машина,|2 телевизора,|микроволновка,|пылесос,|утюг,|посуда,] новая мебель. Все {необходимые документы|документы}. В {этой|данной} квартире {очень|достаточно} [уютно,|тихо,] комфортно. {В шаговой доступности|Рядом|Рядом с домом} [2 аптеки,|3 супермаркета,] 2 стоянки для {авто|автомобилей}."
k = []
rez = []
for x in text:
    if x !='{' and x != '}' and x != ']' and x != '[':
      k.append (x)
k = "".join(k)
x = k.split (" ")
for d in x:
  if '|' in d:
    xxx = d.split ("|")
    idx=random.randint(0,len(xxx)-1)
    rez.append(xxx[idx])      
  else:
    rez.append(d)
print(' '.join(rez))
